WinRAR
	-- ;スイッチの走査を中止
	-@[+] ; ファイルリストの無効化 [有効化]
	-ac ; 圧縮、解凍後にアーカイブ属性をクリア
	-ad[1,2] ; 別の解凍先パス
	-ag[format] ;; 現在日時から書庫名を生成
	-ai ; ファイルの属性を無視
	-am[s,r] ; 書庫名と日時 [保存, 復元]
	-ao ; アーカイブ属性を持つファイルを圧縮
	-ap<path> ; 書庫内部のパスを設定
	-as ; 書庫の内容を同期
	-c- ; コメントの表示を無効
	-cfg- ; デフォルトプロファイルと環境変数を無視
	-cl ; ファイル名を小文字に変換
	-cu ; ファイル名を大文字に変換
	-df ; 圧縮後にファイルを削除
	-dh ; 共有ファイルを開く
	-dr ; ファイルをごみ箱に削除
	-ds ; 圧縮するファイルをソートしない
	-dw ; 書庫作成後にファイルを消去
	-e[+]<attr> ; 除外または対象とするファイルの属性を指定
	-ed ; 空のフォルダを格納しない
	-ep ; 名前からパスを除外
	-ep1 ; 名前からベースフォルダを除外
	-ep2 ; フルパスを展開
	-ep3 ; ドライブレターを含むフルパスを展開
	-ep4<path> ; 名前からパスの先頭を除外
	-f ; ファイルを最新に
	-hp[password] ; ファイルヘッダも暗号化
	-ht[b|c] ; ファイルチェックサムにはハッシュタイプ [BLAKE2、CRC32] を選択
	-id[c,d,n,p,q]; メッセージを表示または無効化します
	-ieml[addr] ; 書庫をメールで送信
	-ierr ; すべてのメッセージを標準エラー出力に出力
	-ilog[name] ; エラーログファイルを作成
	-inul ; すべてのメッセージを無効
	-ioff[n] ; 処理完了後 PC を終了
	-isnd[-] ; 通知サウンドを制御
	-iver ; バージョン番号を表示
	-k ; 書庫をロック
	-kb ; 解凍された破損ファイルを残す
	-log[f][=name]; ログ ファイルに名前を書き込む
	-m<0..5> ; 圧縮方式を指定 (0-無圧縮...3-標準...5-最高圧縮)
	-ma[4|5] ; 書庫形式のバージョンを特定
	-mc<par> ; 高度な圧縮パラメータを指定
	-md<n>[k,m,g] ; KB、MB、または GB の辞書サイズを選択
	-me[par] ; 暗号化パラメータを設定
	-ms[ext;ext] ; 圧縮しないファイルの種類を指定
	-mt<threads> ; スレッドの数をセット
	-n<file> ; ファイルを含んでさらにフィルタ
	-n@ ; 追加のフィルタマスクを標準入力から読み取る
	-n@<list> ; 追加のフィルタマスクをリストファイルから読み取る
	-o[+|-] ; 上書きモードを設定
	-oc ; NTFS 圧縮属性をセット
	-oh ; ファイルの代わりにハードリンクを関連付けて保存
	-oi[0-4][:min]; 同一のファイルを参考文献として保存
	-ol[a] ; [絶対パス] のリンクとしてシンボリックリンクを処理
	-oni ; 互換性に問題のある名前を許可
	-op<path> ; 解凍されたファイルを出力するパスを設定
	-or ; 自動でファイルの名前を変更
	-os ; NTFS ストリームを保存
	-ow ; ファイルセキュリティ情報を処理
	-p[password] ; パスワードを指定
	-qo[-|+] ; クイックオープンインフォメーションを追加 ( 無 | 強制)
	-r ; サブフォルダを再帰的に処理
	-r- ; 再帰的処理を行わない
	-r0 ; ワイルドカードの名前のみサブフォルダを再帰的に処理
	-ri<P>[:<S>] ; 優先度 (0-デフォルト,1-優先度最低..15-優先度最高) およびスリープタイムをセット
	-rr[N] ; リカバリレコードを付加
	-rv[N] ; リカバリボリュームを作成
	-s[<N>,v[-],e]; ソリッド書庫を作成
	-s- ; ソリッド圧縮をしない
	-sc<chr>[obj] ; 文字セットを指定
	-sfx[name] ; 自己解凍書庫を作成
	-si[name] ; 標準入力からデータを読み込む
	-sl<size> ; 指定より小さいサイズのファイルを処理
	-sm<size> ; 指定より大きいサイズのファイルを処理
	-t ; 圧縮後にファイルをテスト
	-ta[mcao]<d> ; YYYYMMDDHHMMSS フォーマットで指定した日時より後に更新されたファイルを処理
	-tb[mcao]<d> ; YYYYMMDDHHMMSS フォーマットで指定した日時より前に更新されたファイルを処理
	-tk ; オリジナルの書庫日時を保つ
	-tl ; 書庫日時を最も新しいファイルにセット
	-tn[mcao]<t> ; 指定した日時より新しいファイルを処理
	-to[mcao]<t> ; 指定した日時より古いファイルを処理
	-ts[m,c,a,p] ; ファイル日時 (更新日時、作成日時、アクセス日時、オリジナルを保持) の格納と復元
	-u ; ファイルを更新
	-v<size>[k,b] ; 指定サイズでボリュームを作成
	-vd ; ボリュームを作成する前にディスクの内容を消去
	-ver[n] ; ファイルの以前の版を残す
	-vn ; 旧式のボリューム名方式を使用
	-vp ; 各ボリュームの前に一時停止
	-w<path> ; ワークディレクトリを割り当て
	-x<file> ; 指定したファイルを除外
	-x@ ; 除外するファイル名を標準入力から読み取る
	-x@<list> ; 指定したリストファイル内のファイルを除外
	-y ; すべての質問に 「はい」
	-z[file] ; ファイルから書庫コメントを付加
